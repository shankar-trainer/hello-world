	update statement 

       make changes in  the existing data
        
		 update table set fieldname=value where condition 
		               if no where then it will update all record 
					   
            		
	   mysql> update student set dob='1995-11-20', name='srinivas kumar' where roll=50001;
	   
	   making calculation in table 
	   
	   mysql> select roll,name,subject,marks,(marks+marks*5/100), dob from student;
	   
	   mysql> select roll,name,subject,marks,(marks+marks*5/100) as 'updated marks', dob from student;
	   
	   
	   sort the record by using order by 
	       by default it is ascending order 
		   
	     
	   mysql> select * from student order by name;
	   mysql> select * from student order by name asc; // same result 
	   
	   descending order 
	   
	   mysql> select * from student order by name desc; // same result

		  equi join  -- it will give you records that is present in both the tables
		  
		  
		    old syntax
		  mysql> select * from student, address where student.roll=address.roll;
		  
		  new sql syntax 
		  
		  mysql> select * from student inner join address on student.roll=address.roll;
		  
		  
		  using table name 
		  
		  mysql> select student.roll, student.name, student.subject, student.marks,student.dob, address.sector,address.city,address.state,address.pin  from student inner join address on student.roll=address.roll
		  
		  
		  left join -- display all values which are present in left table 
		  
		  
		  mysql> select student.roll, student.name, student.subject, student.marks,student.dob, address.sector,address.city,address.state,address.pin  from student left join address on student.roll=address.roll
    
	     right join - display all values which are present in right table 
		 
		 
		 mysql> select student.roll, student.name, student.subject, student.marks,student.dob, address.sector,address.city,address.state,address.pin  from student right  join address on student.roll=address.roll
