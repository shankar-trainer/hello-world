PROGRAM "BUBBLE SORT"

 PRINT "ENTER ARRAY SIZE"
 INPUT N
  AR(N)=[]

 FOR I=0 TO N-1
   PRINT "ENTER NO"
   READ  AR(I)
 END FOR 
 PRINT "UNSORTED ARRAY "
 FOR I=0 TO N-1
   PRINT AR(I)
 END FOR 
  
 FOR I=0 TO N-1
   FOR J=0 TO N-2-I
    IF AR(J)>AR(J+1)
      TEMP=AR(J)
      AR(J)=AR(J+1)
      AR(J+1)=TEMP
    END IF 
   END FOR
 END FOR  

 PRINT "SORTED ARRAY "
  FOR I=0 TO N-1
   PRINT AR(I)
  END FOR 
 END
