Program "insertion sort"

 print "enter array length"
 read n
 
 ar(n), key
  
  For i=0 to n-1
    Print "enter element"
	read ar(i)
  End For 
  
  Print "unsorted array "
  
  For i=0 to n-1
      print ar(i)
  End For 
   // insertion sorting 
  For j=1 to n-1 
    key=ar(j)
    i=j-1

   while i>0 and ar(i)<key 
      ar(i+1)=ar(i)
      i=i-1 
   end while 
      ar(i+1)=key 
  End For
 
  Print "sorted array "
  
  For i=0 to n-1
      print ar(i)
  End For  

END
