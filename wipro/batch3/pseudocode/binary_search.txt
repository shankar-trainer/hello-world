 sub binary_search(ar, key, high, low, search_index)
   sort(ar)
   
  while low<high 
    mid=(low+high)/2
	if key==ar(mid)
	   search_index=mid 
	else if  key>ar(mid)
      low=mid+1
     else 
       high=mid-1	 
  end while 
  return search_index
  end Sub
