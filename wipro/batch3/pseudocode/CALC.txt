START "CALCULATION PRGRAM"

  OUTPUT "ENTER TWO NO"
  READ A,B

  C=A+B
  D=A-B
  E=A*B
  F=A/B
  G=A MOD B
  H=A ** B

  OUTPUT "NO1 IS ",A
  OUTPUT "NO2 IS ",B
  OUTPUT "ADDITION  IS ",C
  OUTPUT "SUBTRACTION  IS ",D
  OUTPUT "MULTIPLICATION  IS ",E
  OUTPUT "DIVISION IS  ",F
  OUTPUT "MODULUS  IS ",G
  OUTPUT "EXPONENT  IS ",H

END
