Function has name , parameter , return type 

Types 

   Function no parameter no return 

           BEGIN 
			   FUNCTION hello()

			   End  FUNCTION

			  hello()
           END
		   
		     OUTPUT

	       BEGIN
            FUNCTION rectangle()
			 INPUT W,B
		     COMPUTE AREA<-W*B
            PRINT "AREA IS ", AREA			 
		   End  FUNCTION
		   
		   rectangle()
		   rectangle()
	       rectangle()
	      END
		  
		  OUTPUT
		   2 3--> 6

	   Function with parameter and no return
	   
	    BEGIN 
		  Declare amount,rate as double
		  Declare time as Integer 
		  
		  FUNCTION interest_calculation(a,r,t)
			  COMPUTE I<-a*r*t
			  PRINT "INTEREST IS ",I
          END FUNCTION 

          interest_calculation(1000,0.05,5)	
          
          PRINT "Enter amount rate and time "
          INPUT amount, rate, time

          interest_calculation(amount,rate,time) 		  
		END

      Function with parameter and return
	  
	  BEGIN
		DECLRAE a,b,c as Integer 
		 FUNCTION addition(x,y)
		   s=x+y
		   RETURN s;
		 END FUNCTION

		 PRINT("ENTER TWO NUMBER")
		 INPUT a,b
		 c=addition(a,b)
		 PRINT "SUM OF ",a," AND ",b," IS ", c
      END
      OUTPUT
	    ENTER TWO NUMBER
		23,34 
		output --
		SUMOF 23 AND 34 IS 57
