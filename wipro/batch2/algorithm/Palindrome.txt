 Read n
 n1=n
 k=0
 
 While n<>0 Loop 
  k=k*10+n%10
  n=n/10  
 End Loop 

 If n1=k then 
  Print "No is palindrome "
 Else
  Print "No is not palindrome"
 End If 
End
