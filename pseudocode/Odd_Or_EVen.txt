PSEUDOCODE
------------------------------------
BEGIN
 Declare n as Integer 

 PRINT("Enter Number")
 INPUT n
 
 IF(n Mod 2=0) THEN
 PRINT("EVEN")
 ELSE
 PRINT("ODD")
 ENDIF

END

      ENDIF
