	Function series(n)
	 Declare n1,n2,res as Integer

fibonacci series using recursion

Begin
 Declare n as Integer
	Function series(n)
	 if n=0 
	  return 0
	 end if 
	 if n=1
      return 1
	 end if  
	 return series(n-1)+series(n-2)
	End Function

   n=5
